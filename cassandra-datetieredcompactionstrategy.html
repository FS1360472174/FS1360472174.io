//timeUnit 时间间隔，base 最小时间 ,now 当前时间
@VisibleForTesting
static List> getBuckets(Collection> files, long timeUnit, int base, long now, long maxWindowSize)
{
// Sort files by age. Newest first.
final List> sortedFiles = Lists.newArrayList(files);
Collections.sort(sortedFiles, Collections.reverseOrder(new Comparator>()
{
public int compare(Pair p1, Pair p2)
{
return p1.right.compareTo(p2.right);
}
}));

    List<List<T>> buckets = Lists.newArrayList();
    //target.size=now
    //target.divPosition=timeUnit
    Target target = getInitialTarget(now, timeUnit, maxWindowSize);
    // API https://google.github.io/guava/releases/19.0/api/docs/com/google/common/collect/PeekingIterator.html#peek()
    PeekingIterator<Pair<T, Long>> it = Iterators.peekingIterator(sortedFiles.iterator());

    outerLoop:
    while (it.hasNext())
    {
           //target.onTarget(),将target.divPosition与集合中的值/size 进行比较
           while (!target.onTarget(it.peek().right))
        {
            // If the file is too new for the target, skip it.
            if (target.compareToTimestamp(it.peek().right) < 0)
            {
                it.next();

                if (!it.hasNext())
                    break outerLoop;
            }
            else // If the file is too old for the target, switch targets.
                //将target.size 增大，或者减少divPosition
                target = target.nextTarget(base);
        }
        List<T> bucket = Lists.newArrayList();
        while (target.onTarget(it.peek().right))
        {
            bucket.add(it.next().left);

            if (!it.hasNext())
                break;
        }
        buckets.add(bucket);
    }

    return buckets;
}

testFilterOldSSTables

//将sstable的max-timestamp 
@VisibleForTesting
static Iterable<SSTableReader> filterOldSSTables(List<SSTableReader> sstables, long maxSSTableAge, long now)
{
    if (maxSSTableAge == 0)
        return sstables;
    final long cutoff = now - maxSSTableAge;
    return filter(sstables, new Predicate<SSTableReader>()
    {
        @Override
        public boolean apply(SSTableReader sstable)
        {
            return sstable.getMaxTimestamp() >= cutoff;
        }
    });
}
