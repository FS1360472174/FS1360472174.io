3.register MBean

public class Agent {
    public static void main(String[] args)
            throws MalformedObjectNameException, NullPointerException, InstanceAlreadyExistsException,
            MBeanRegistrationException, NotCompliantMBeanException, InterruptedException {
        MBeanServer server = ManagementFactory.getPlatformMBeanServer();
        ObjectName name = new ObjectName("com.fs.metrics:type=CrossDCLatency"); 
        Latency mbean = new Latency();
        server.registerMBean(mbean, name);

        System.out.println("waiting here for remote management...");
        Thread.sleep(Long.MAX_VALUE);
    }
}
